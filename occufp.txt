
model {
# Priors
  psi ~ dunif(0, 1)
  p ~ dunif(0, 1)
  fp ~ dunif(0, 1)
# Likelihood and process model
  for (i in 1:nsites) { # Loop over sites
    z[i] ~ dbern(psi) # State model
    for (j in 1:nsurveys) { # Loop over replicate surveys
      y[i,j] ~ dbern(z[i]*p + (1-z[i])*fp) # Observation model
    }
  }
}
